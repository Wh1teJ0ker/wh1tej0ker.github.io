from gmpy2 import *

def N_Common(c1, c2, e1, e2, n):
    t, s1, s2 = gcdext(e1, e2)

    m = powmod(c1, s1, n) * powmod(c2, s2, n) % n
    return long_to_bytes(iroot(m, t)[0])
